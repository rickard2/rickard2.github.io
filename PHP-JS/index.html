PHP 5.5

Generators


                function even($max = 10) {
                    $value = 0;
                    
                    while ($value <= $max) {
                        yield $value;
                        $value += 2;
                    }
                }
                
                $generator = even(8);
                
                foreach ($generator as $number) {
                    // 2, 4, 6, 8
                }
